\newglossaryentry{discriminant}{
  name = {判别式},
  description = {\emph{discriminant} $\Delta = b^2-4ac$ 用来确定二次函数的根的个数的表达式，当$\Delta > 0$时，二次方程有两个不一样的实根；当$\Delta = 0$时，二次方程有两个相等的实根，或者称之为一个根；$\Delta < 0$时，二次方程没有实数解，但是有两个共轭的复数解}
}

  name = {根},

}
